from abc import ABC, abstractmethod
from typing import Any

class BaseAIModel(ABC):
    vram_mb: int = 0

    @abstractmethod
    def load(self) -> None: ...

    @abstractmethod
    def predict(self, *args, **kwargs) -> Any: ...

    def unload(self) -> None:
        pass

    @property
    @abstractmethod
    def is_loaded(self) -> bool: ...
```

import threading
from typing import Dict
from .base import BaseAIModel

class ModelRegistry:
    def __init__(self) -> None:
        self._models: Dict[str, BaseAIModel] = {}
        self._lock = threading.Lock()

    def register(self, name: str, model: BaseAIModel) -> None:
        with self._lock:
            self._models[name] = model

    def get(self, name: str) -> BaseAIModel:
        with self._lock:
            if name not in self._models:
                raise KeyError(f"Model '{name}' is not registered")
            model = self._models[name]
            if not model.is_loaded:
                model.load()
            return model

registry = ModelRegistry()
```

from pathlib import Path
from typing import Any
import torch
import timm
from PIL import Image
from torchvision import transforms
from ..base import BaseAIModel
from .classes import CLASSES, get_treatment

_TRANSFORM = transforms.Compose([
    transforms.Resize(256),
    transforms.CenterCrop(224),
    transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225]),
])

class DiseaseModel(BaseAIModel):
    vram_mb = 300

    def __init__(self, checkpoint_path: str) -> None:
        self._checkpoint = Path(checkpoint_path)
        self._model: Any = None
        self._device = "cuda" if torch.cuda.is_available() else "cpu"

    def load(self) -> None:
        self._model = timm.create_model("efficientnet_v2_s", pretrained=False, num_classes=38)
        if self._checkpoint.exists():
            state = torch.load(self._checkpoint, map_location="cpu", weights_only=True)
            self._model.load_state_dict(state)
        self._model.train(mode=False)
        self._model = self._model.to(self._device)

    @property
    def is_loaded(self) -> bool:
        return self._model is not None

    def predict(self, image: Image.Image) -> dict:
        tensor = _TRANSFORM(image.convert("RGB")).unsqueeze(0).to(self._device)
        with torch.no_grad():
            probs = torch.softmax(self._model(tensor), dim=1)[0]
        top5 = probs.topk(5)
        results = [
            {"label": CLASSES[idx], "confidence": round(prob, 4), "treatment": get_treatment(CLASSES[idx])}
            for prob, idx in zip(top5.values.tolist(), top5.indices.tolist())
        ]
        return {"topPrediction": results[0], "alternatives": results[1:]}

    def unload(self) -> None:
        self._model = None
        if torch.cuda.is_available():
            torch.cuda.empty_cache()
```

"""
Run: python -m ai.disease.train
Produces: backend/models/efficientnet_v2s_plantvillage.pth
Estimated time: ~2h on RTX 3050
"""
import torch, timm
from pathlib import Path
from torch import nn, optim
from torch.utils.data import DataLoader, random_split
from torchvision import transforms, datasets

DATA_DIR = Path(__file__).parent.parent.parent / "data" / "plantvillage"
OUT_PATH = Path(__file__).parent.parent.parent / "models" / "efficientnet_v2s_plantvillage.pth"
EPOCHS, BATCH, LR = 20, 32, 1e-4

TRAIN_TF = transforms.Compose([
    transforms.RandomResizedCrop(224), transforms.RandomHorizontalFlip(),
    transforms.ColorJitter(0.2, 0.2, 0.2, 0.1), transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225]),
])
VAL_TF = transforms.Compose([
    transforms.Resize(256), transforms.CenterCrop(224), transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225]),
])

def mixup(x, y, alpha=0.2):
    lam = torch.distributions.Beta(torch.tensor(alpha), torch.tensor(alpha)).sample().item()
    idx = torch.randperm(x.size(0))
    return lam * x + (1 - lam) * x[idx], y, y[idx], lam

def train():
    device = "cuda" if torch.cuda.is_available() else "cpu"
    print(f"Training on {device}")
    full = datasets.ImageFolder(DATA_DIR, transform=TRAIN_TF)
    nv, nt = int(len(full) * 0.1), int(len(full) * 0.1)
    train_ds, val_ds, _ = random_split(full, [len(full) - nv - nt, nv, nt])
    val_ds.dataset.transform = VAL_TF
    tl = DataLoader(train_ds, batch_size=BATCH, shuffle=True, num_workers=4, pin_memory=True)
    vl = DataLoader(val_ds, batch_size=BATCH, num_workers=4, pin_memory=True)
    model = timm.create_model("efficientnet_v2_s", pretrained=True, num_classes=38).to(device)
    opt = optim.AdamW(model.parameters(), lr=LR, weight_decay=1e-2)
    sched = optim.lr_scheduler.CosineAnnealingLR(opt, T_max=EPOCHS)
    crit = nn.CrossEntropyLoss()
    best_acc, patience, no_imp = 0.0, 5, 0
    for epoch in range(EPOCHS):
        model.train()
        for x, y in tl:
            x, y = x.to(device), y.to(device)
            xm, ya, yb, lam = mixup(x, y)
            loss = lam * crit(model(xm), ya) + (1 - lam) * crit(model(xm), yb)
            opt.zero_grad(); loss.backward(); opt.step()
        sched.step()
        model.train(mode=False)
        correct = total = 0
        with torch.no_grad():
            for x, y in vl:
                x, y = x.to(device), y.to(device)
                correct += (model(x).argmax(1) == y).sum().item(); total += y.size(0)
        acc = correct / total
        print(f"Epoch {epoch+1}/{EPOCHS}  val_acc={acc:.4f}")
        if acc > best_acc:
            best_acc = acc; no_imp = 0
            OUT_PATH.parent.mkdir(parents=True, exist_ok=True)
            torch.save(model.state_dict(), OUT_PATH)
            print(f"  Saved (best={best_acc:.4f})")
        else:
            no_imp += 1
            if no_imp >= patience: print("Early stop."); break
    print(f"Done. Best val acc: {best_acc:.4f}")

if __name__ == "__main__":
    train()
```

import torch
import torch.nn as nn

class PatchTST(nn.Module):
    def __init__(self, seq_len=1008, pred_len=5, patch_size=16,
                 d_model=128, n_heads=8, n_layers=3, dropout=0.1):
        super().__init__()
        assert seq_len % patch_size == 0
        self.patch_size = patch_size
        n_patches = seq_len // patch_size
        self.patch_embed = nn.Linear(patch_size, d_model)
        self.pos_embed = nn.Parameter(torch.zeros(1, n_patches, d_model))
        enc = nn.TransformerEncoderLayer(d_model=d_model, nhead=n_heads,
            dim_feedforward=d_model * 4, dropout=dropout,
            batch_first=True, norm_first=True)
        self.encoder = nn.TransformerEncoder(enc, num_layers=n_layers)
        self.head = nn.Linear(d_model * n_patches, pred_len)
        nn.init.trunc_normal_(self.pos_embed, std=0.02)

    def forward(self, x):
        x = x.unfold(-1, self.patch_size, self.patch_size)
        x = self.patch_embed(x) + self.pos_embed
        return self.head(self.encoder(x).flatten(1))
```

from pathlib import Path
import numpy as np
import torch
from ..base import BaseAIModel
from .patch_tst import PatchTST

METRICS = ["temp", "humidity", "ph", "ec"]
SEQ_LEN, PRED_LEN = 1008, 5

class ForecastModel(BaseAIModel):
    vram_mb = 50

    def __init__(self, checkpoints_dir: str) -> None:
        self._dir = Path(checkpoints_dir)
        self._models: dict[str, PatchTST] = {}
        self._scalers: dict[str, tuple[float, float]] = {}

    def load(self) -> None:
        for metric in METRICS:
            model = PatchTST(seq_len=SEQ_LEN, pred_len=PRED_LEN)
            ckpt = self._dir / f"patchtst_{metric}.pth"
            if ckpt.exists():
                state = torch.load(ckpt, map_location="cpu", weights_only=True)
                model.load_state_dict(state["model"])
                self._scalers[metric] = (state["min"], state["max"])
            else:
                self._scalers[metric] = (0.0, 1.0)
            model.train(mode=False)
            self._models[metric] = model

    @property
    def is_loaded(self) -> bool:
        return len(self._models) == len(METRICS)

    def predict(self, readings: list[float], metric: str) -> list[float]:
        if metric not in self._models:
            raise ValueError(f"Unknown metric {metric!r}. Valid: {METRICS}")
        arr = np.array(readings[-SEQ_LEN:], dtype=np.float32)
        if len(arr) < SEQ_LEN:
            arr = np.pad(arr, (SEQ_LEN - len(arr), 0), mode="edge")
        min_v, max_v = self._scalers[metric]
        span = max_v - min_v if max_v != min_v else 1.0
        tensor = torch.from_numpy((arr - min_v) / span).unsqueeze(0)
        with torch.no_grad():
            out = self._models[metric](tensor)[0].tolist()
        return [round(float(v * span + min_v), 3) for v in out]
```

### Task 9: Forecast training script

**Files:** Create `backend/ai/forecast/train.py`

- [ ] **Step 1: Create `backend/ai/forecast/train.py`**

```python
"""
python -m ai.forecast.train --phase 1           # synthetic (~1h)
python -m ai.forecast.train --phase 2 --room-id ROOM_ID  # Firestore
Produces: backend/models/patchtst_{metric}.pth for temp, humidity, ph, ec
"""
import argparse, asyncio, math
from pathlib import Path
import numpy as np
import torch
from torch import nn, optim
from torch.utils.data import DataLoader, TensorDataset
from .patch_tst import PatchTST

METRICS = ["temp", "humidity", "ph", "ec"]
SEQ_LEN, PRED_LEN = 1008, 5
OUT_DIR = Path(__file__).parent.parent.parent / "models"

def _synthetic(metric, n_days=90):
    ticks = n_days * 144
    t = np.arange(ticks, dtype=np.float32)
    rng = np.random.default_rng(42)
    if metric == "temp":
        return (24.0 + 2.0 * np.sin(2 * math.pi * t / 144) + rng.normal(0, 0.3, ticks)).tolist()
    elif metric == "humidity":
        return (65.0 - 5.0 * np.sin(2 * math.pi * t / 144) + rng.normal(0, 1.0, ticks)).tolist()
    elif metric == "ph":
        return (np.cumsum(rng.normal(0.001, 0.01, ticks)) % 1.5 + 5.8).tolist()
    return (1.8 + 0.3 * np.sin(2 * math.pi * t / (144 * 7)) + rng.normal(0, 0.05, ticks)).tolist()

def _windows(series):
    arr = np.array(series, dtype=np.float32)
    min_v, max_v = float(arr.min()), float(arr.max())
    span = max_v - min_v if max_v != min_v else 1.0
    daily = ((arr - min_v) / span).reshape(-1, 144).mean(axis=1)
    wd = SEQ_LEN // 144
    X, y = [], []
    for i in range(len(daily) - wd - PRED_LEN + 1):
        X.append(np.repeat(daily[i:i + wd], 144)[:SEQ_LEN])
        y.append(daily[i + wd:i + wd + PRED_LEN])
    return np.array(X), np.array(y), min_v, max_v

def _train(metric, series):
    X, y, mn, mx = _windows(series)
    loader = DataLoader(TensorDataset(torch.from_numpy(X), torch.from_numpy(y)),
                        batch_size=64, shuffle=True)
    device = "cuda" if torch.cuda.is_available() else "cpu"
    model = PatchTST(seq_len=SEQ_LEN, pred_len=PRED_LEN).to(device)
    opt = optim.Adam(model.parameters(), lr=1e-3)
    sched = optim.lr_scheduler.CosineAnnealingLR(opt, T_max=50)
    crit = nn.MSELoss()
    best, best_state = float("inf"), None
    for epoch in range(50):
        model.train()
        total = 0.0
        for xb, yb in loader:
            xb, yb = xb.to(device), yb.to(device)
            loss = crit(model(xb), yb)
            opt.zero_grad(); loss.backward(); opt.step()
            total += loss.item()
        sched.step()
        avg = total / len(loader)
        if avg < best:
            best = avg
            best_state = {k: v.clone() for k, v in model.state_dict().items()}
        if (epoch + 1) % 10 == 0:
            print(f"  [{metric}] epoch {epoch+1}/50 loss={avg:.6f}")
    OUT_DIR.mkdir(parents=True, exist_ok=True)
    torch.save({"model": best_state, "min": mn, "max": mx}, OUT_DIR / f"patchtst_{metric}.pth")
    print(f"  Saved patchtst_{metric}.pth (best={best:.6f})")

async def _firestore(room_id, metric):
    import sys; sys.path.insert(0, str(Path(__file__).parent.parent.parent))
    from firebase_client import get_sync_db
    db = get_sync_db()
    return [d.to_dict().get(metric, 0.0) for d in
            db.collection("rooms").document(room_id)
            .collection("readings").order_by("timestamp").stream()]

def main():
    p = argparse.ArgumentParser()
    p.add_argument("--phase", type=int, choices=[1, 2], default=1)
    p.add_argument("--room-id", type=str, default=None)
    args = p.parse_args()
    for metric in METRICS:
        print(f"\nTraining PatchTST for: {metric}")
        if args.phase == 1:
            series = _synthetic(metric)
        else:
            if not args.room_id: raise ValueError("--room-id required")
            series = asyncio.run(_firestore(args.room_id, metric))
            if len(series) < SEQ_LEN + PRED_LEN * 144:
                print(f"  Not enough data ({len(series)} ticks). Skipping."); continue
        _train(metric, series)

if __name__ == "__main__": main()
```

from pathlib import Path
from typing import Optional
import chromadb
from sentence_transformers import SentenceTransformer

KNOWLEDGE_DIR = Path(__file__).parent / "knowledge"
DEFAULT_CHROMA_PATH = Path(__file__).parent.parent.parent / "data" / "chroma"

class GreenhouseRAG:
    def __init__(self, chroma_path: Optional[str] = None) -> None:
        self._path = Path(chroma_path) if chroma_path else DEFAULT_CHROMA_PATH
        self._client = None
        self._collection = None
        self._embedder: Optional[SentenceTransformer] = None

    @property
    def is_ready(self) -> bool:
        return self._collection is not None

    def build(self) -> None:
        self._embedder = SentenceTransformer("all-MiniLM-L6-v2")
        self._path.mkdir(parents=True, exist_ok=True)
        self._client = chromadb.PersistentClient(path=str(self._path))
        self._collection = self._client.get_or_create_collection(
            "greenhouse_knowledge", metadata={"hnsw:space": "cosine"})
        docs, ids = [], []
        for p in sorted(KNOWLEDGE_DIR.glob("*.md")):
            for j, chunk in enumerate(self._chunk(p.read_text(encoding="utf-8"))):
                docs.append(chunk); ids.append(f"{p.stem}_{j}")
        if docs:
            self._collection.upsert(
                documents=docs,
                embeddings=self._embedder.encode(docs, show_progress_bar=False).tolist(),
                ids=ids)

    def load(self) -> None:
        self._embedder = SentenceTransformer("all-MiniLM-L6-v2")
        self._client = chromadb.PersistentClient(path=str(self._path))
        self._collection = self._client.get_collection("greenhouse_knowledge")

    def retrieve(self, query: str, n: int = 3) -> list[str]:
        if not self.is_ready: return []
        emb = self._embedder.encode([query], show_progress_bar=False).tolist()
        return self._collection.query(query_embeddings=emb, n_results=n)["documents"][0]

    @staticmethod
    def _chunk(text: str, size: int = 400, overlap: int = 50) -> list[str]:
        words = text.split(); chunks, start = [], 0
        while start < len(words):
            chunks.append(" ".join(words[start:start + size])); start += size - overlap
        return chunks
```

from dataclasses import dataclass
from typing import Literal

Severity = Literal["info", "warning", "critical"]

@dataclass
class RuleAction:
    metric: str
    action: str
    severity: Severity
    current: float
    target_min: float
    target_max: float

def evaluate_rules(metrics: dict, targets: dict) -> list[RuleAction]:
    out: list[RuleAction] = []

    ph = float(metrics.get("ph", 7.0))
    ph_min, ph_max = targets.get("ph", [5.8, 6.5])
    if ph > ph_max + 0.2:
        out.append(RuleAction("ph", f"Dose {max(1, round((ph - ph_max) * 10))}ml pH Down",
                              "warning", ph, ph_min, ph_max))
    elif ph < ph_min - 0.2:
        out.append(RuleAction("ph", f"Dose {max(1, round((ph_min - ph) * 10))}ml pH Up",
                              "warning", ph, ph_min, ph_max))

    ec = float(metrics.get("ec", 1.5))
    ec_min, ec_max = targets.get("ec", [1.2, 2.0])
    if ec < ec_min - 0.3:
        out.append(RuleAction("ec", "Apply nutrient solution", "warning", ec, ec_min, ec_max))
    elif ec > ec_max + 0.3:
        out.append(RuleAction("ec", "Flush with plain water", "warning", ec, ec_min, ec_max))

    temp = float(metrics.get("temp", 24.0))
    t_min, t_max = targets.get("temp", [20.0, 28.0])
    if temp > t_max + 2.0:
        out.append(RuleAction("temp", "Reduce room temperature", "critical", temp, t_min, t_max))
    elif temp < t_min - 2.0:
        out.append(RuleAction("temp", "Increase room temperature", "critical", temp, t_min, t_max))

    hum = float(metrics.get("humidity", 60.0))
    h_min, h_max = targets.get("humidity", [50.0, 70.0])
    if hum > h_max + 5.0:
        out.append(RuleAction("humidity", "Increase ventilation / lower humidity",
                              "warning", hum, h_min, h_max))
    elif hum < h_min - 5.0:
        out.append(RuleAction("humidity", "Add humidification", "warning", hum, h_min, h_max))

    vpd = float(metrics.get("vpd", 1.0))
    if vpd > 1.6:
        out.append(RuleAction("vpd", "Raise humidity or lower temperature", "warning", vpd, 0.4, 1.6))
    elif vpd < 0.4:
        out.append(RuleAction("vpd", "Lower humidity or raise temperature", "warning", vpd, 0.4, 1.6))

    return out
```

```python
from __future__ import annotations
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Optional
from ..base import BaseAIModel
from .rules import evaluate_rules

LGBM_PATH = Path(__file__).parent.parent.parent / "models" / "recommend_lgbm.joblib"
STAGE_MAP = {"seedling": 0, "veg": 1, "vegetative": 1, "flower": 2, "flowering": 2, "flush": 3}

class RecommendModel(BaseAIModel):
    vram_mb = 0

    def __init__(self) -> None:
        self._lgbm: Optional[Any] = None
        self._loaded = False

    def load(self) -> None:
        if LGBM_PATH.exists():
            import joblib
            self._lgbm = joblib.load(LGBM_PATH)
        self._loaded = True

    @property
    def is_loaded(self) -> bool:
        return self._loaded

    @property
    def lgbm_active(self) -> bool:
        return self._lgbm is not None

    def predict(self, room: dict) -> dict:
        metrics = room.get("currentMetrics", {})
        targets = room.get("targets", {})
        stage = str(room.get("stage", "veg"))
        day = int(room.get("day", 1))
        total_days = max(int(room.get("totalDays", 60)), 1)

        immediate = [
            {"metric": a.metric, "action": a.action, "severity": a.severity,
             "current": a.current, "target_min": a.target_min, "target_max": a.target_max}
            for a in evaluate_rules(metrics, targets)
        ]

        optimizations = []
        if self._lgbm is not None:
            feats = self._features(metrics, targets, stage, day, total_days)
            try:
                deltas = self._lgbm.predict([feats])[0]
                for i, key in enumerate(["temp", "humidity", "ph", "ec"]):
                    delta = float(deltas[i])
                    if abs(delta) > 0.05:
                        cr = targets.get(key, [0.0, 0.0])
                        optimizations.append({
                            "setpoint": key, "current": cr,
                            "suggested": [round(cr[0] + delta, 2), round(cr[1] + delta, 2)],
                            "confidence": round(min(0.95, 0.5 + abs(delta) * 0.3), 2),
                            "reason": f"7-day trend analysis suggests {delta:+.1f} adjustment",
                        })
            except Exception:
                pass

        return {
            "immediate": immediate, "optimizations": optimizations,
            "model_active": self.lgbm_active,
            "generated_at": datetime.now(timezone.utc).isoformat(),
        }

    @staticmethod
    def _features(metrics, targets, stage, day, total_days) -> list[float]:
        f = []
        for key in ["temp", "humidity", "ph", "ec", "vpd"]:
            v = float(metrics.get(key, 0.0))
            t = targets.get(key, [v, v])
            f += [v, v - (float(t[0]) + float(t[1])) / 2.0]
        f += [float(STAGE_MAP.get(stage.lower(), 1)), day / total_days]
        return f
```

    }

  } finally {

  }
};

### Task 18: LightGBM training script

**Files:** Create `backend/ai/recommend/train.py`

> Run this after 30+ days of real Firestore readings. Until then, the rule engine handles all recommendations.

- [ ] **Step 1: Create `backend/ai/recommend/train.py`**

```python
"""
python -m ai.recommend.train --room-id ROOM_ID
Produces: backend/models/recommend_lgbm.joblib
Requires 30+ days of Firestore readings.
"""
import argparse, asyncio
from pathlib import Path
import numpy as np

OUT_PATH = Path(__file__).parent.parent.parent / "models" / "recommend_lgbm.joblib"
STAGE_MAP = {"seedling":0,"veg":1,"vegetative":1,"flower":2,"flowering":2,"flush":3}
SENSORS = ["temp","humidity","ph","ec","vpd"]
TARGETS_KEYS = ["temp","humidity","ph","ec"]

async def _fetch(room_id):
    import sys; sys.path.insert(0, str(Path(__file__).parent.parent.parent))
    from firebase_client import get_sync_db
    db = get_sync_db()
    readings = [d.to_dict() for d in
        db.collection("rooms").document(room_id).collection("readings").order_by("timestamp").stream()]
    room = db.collection("rooms").document(room_id).get().to_dict() or {}
    return readings, room

def _dataset(readings, room):
    tgts = room.get("targets", {})
    stage_enc = float(STAGE_MAP.get(str(room.get("stage","veg")).lower(), 1))
    total = max(int(room.get("totalDays", 60)), 1)
    window = 144 * 7
    X, y = [], []
    for i in range(window, len(readings) - 144):
        chunk = readings[i - window:i]
        future = readings[i:i + 144]
        feats = []
        for key in SENSORS:
            vals = np.array([float(r.get(key, 0)) for r in chunk])
            t = tgts.get(key, [float(vals.mean())] * 2)
            center = (float(t[0]) + float(t[1])) / 2.0
            slope = float(np.polyfit(range(len(vals)), vals, 1)[0])
            feats += [float(vals.mean()), float(vals.std()), slope, float(vals.mean()) - center]
        feats += [stage_enc, i / len(readings)]
        X.append(feats)
        labels = []
        for key in TARGETS_KEYS:
            fm = float(np.mean([float(r.get(key, 0)) for r in future]))
            cm = float(np.mean([float(r.get(key, 0)) for r in chunk[-144:]]))
            t = tgts.get(key, [cm] * 2)
            labels.append((float(t[0]) + float(t[1])) / 2.0 - cm)
        y.append(labels)
    return np.array(X, dtype=np.float32), np.array(y, dtype=np.float32)

def train(room_id):
    import joblib
    import lightgbm as lgb
    from sklearn.multioutput import MultiOutputRegressor
    print(f"Fetching readings for room {room_id}...")
    readings, room = asyncio.run(_fetch(room_id))
    if len(readings) < 144 * 30:
        print(f"Need 30+ days ({len(readings)//144:.1f} found). Exiting."); return
    print(f"Building dataset from {len(readings)} readings...")
    X, y = _dataset(readings, room)
    print(f"Dataset: {X.shape[0]} samples, {X.shape[1]} features, {y.shape[1]} targets")
    model = MultiOutputRegressor(
        lgb.LGBMRegressor(n_estimators=300, learning_rate=0.05, num_leaves=31, random_state=42))
    model.fit(X, y)
    OUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    joblib.dump(model, OUT_PATH)
    print(f"Saved to {OUT_PATH}")

if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--room-id", required=True)
    train(p.parse_args().room_id)
```
